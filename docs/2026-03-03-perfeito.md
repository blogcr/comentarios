<script type='text/javascript'>
//<![CDATA[
(function() {

  const scriptURL = "https://script.google.com/macros/s/AKfycbxNRrY5FChcSaYwGaoCsuKso5cz1ATEfjn8ySlmK_V60lNlUWmLTQ8AsibNvKCYlFsX/exec";

  const form = document.getElementById("formComentario");
  const lista = document.getElementById("comentarios");
  const contador = document.getElementById("contadorComentarios");

  if (!form || !lista) return;

  /* ===== URL CANÔNICA DO POST ===== */
  const canonicalTag = document.querySelector("link[rel='canonical']");
  const postId = canonicalTag
    ? canonicalTag.href.replace(/\/$/, "")
    : window.location.href.split("?")[0].replace(/\/$/, "");

  const tituloPost = document.title.split('|')[0].trim();

  const marcador = document.querySelector(".post_tag a")
    ? document.querySelector(".post_tag a").innerText.trim()
    : "";

  function formatarData(dataISO) {
    const agora = new Date();
    const data = new Date(dataISO);
    const diff = Math.floor((agora - data) / 1000);

    if (diff < 60) return "agora mesmo";
    if (diff < 3600) {
      const m = Math.floor(diff / 60);
      return "há " + m + (m === 1 ? " minuto" : " minutos");
    }
    if (diff < 86400) {
      const h = Math.floor(diff / 3600);
      return "há " + h + (h === 1 ? " hora" : " horas");
    }
    return data.toLocaleDateString("pt-BR");
  }

  function renderizarComentario(item) {

    const div = document.createElement("div");
    div.className = "comentario-item";

    if (item.parentId) {
      div.setAttribute("data-parent", "true");
    }

    div.innerHTML =
      "<div class='comentario-cabecalho'>" +
        "<strong>" + item.nome + "</strong>" +
        "<span>" + formatarData(item.data) + "</span>" +
      "</div>" +
      "<div class='comentario-texto'>" +
        item.comentario +
      "</div>" +
      (!item.parentId
        ? "<div class='comentario-acoes'>" +
            "<button class='botao-resposta-inline'>Responder</button>" +
          "</div>"
        : "");

    if (!item.parentId) {
      const botao = div.querySelector(".botao-resposta-inline");

      botao.onclick = function() {

        const existente = div.querySelector(".resposta-box");
        if (existente) {
          existente.remove();
          return;
        }

        const box = document.createElement("div");
        box.className = "resposta-box";

        box.innerHTML =
          "<input class='input-resposta' placeholder='Seu nome' />" +
          "<textarea class='textarea-resposta' placeholder='Escreva sua resposta'></textarea>" +
          "<button class='botao-resposta'>Enviar</button>";

        box.querySelector(".botao-resposta").onclick = function() {

          const nome = box.querySelector(".input-resposta").value.trim();
          const comentario = box.querySelector(".textarea-resposta").value.trim();

          if (!nome || !comentario) return;

          fetch(scriptURL, {
            method: "POST",
            body: JSON.stringify({
              postId: postId,
              titulo: tituloPost,
              autorPost: marcador,
              nome: nome,
              comentario: comentario,
              ativo: true,
              parentId: item.linha
            })
          }).then(function(){
            carregarComentarios();
          });
        };

        div.appendChild(box);
      };
    }

    return div;
  }

  function carregarComentarios() {

    fetch(scriptURL)
      .then(function(r){ return r.json(); })
      .then(function(dados){

        const comentariosPost = dados.filter(function(c){
          return (
            c.postId &&
            c.postId.replace(/\/$/, "") === postId &&
            c.ativo === true
          );
        });

        comentariosPost.sort(function(a,b){
          return new Date(a.data) - new Date(b.data);
        });

        lista.innerHTML = "";

        comentariosPost.forEach(function(c){
          lista.appendChild(renderizarComentario(c));
        });

        if (contador) {
          contador.innerHTML =
            comentariosPost.length === 0
              ? "Seja o primeiro a comentar."
              : comentariosPost.length +
                (comentariosPost.length === 1
                  ? " comentário"
                  : " comentários");
        }

      });
  }

  carregarComentarios();

  form.onsubmit = function(e){
    e.preventDefault();

    const nome = document.getElementById("nome").value.trim();
    const comentario = document.getElementById("comentario").value.trim();

    if (!nome || !comentario) return;

    fetch(scriptURL, {
      method: "POST",
      body: JSON.stringify({
        postId: postId,
        titulo: tituloPost,
        autorPost: marcador,
        nome: nome,
        comentario: comentario,
        ativo: true,
        parentId: null
      })
    }).then(function(){
      form.reset();
      carregarComentarios();
    });
  };

})();
 //]]>
</script>
